import Link from "next/link";
import { notFound } from "next/navigation";

// Change this import if your project uses a different prisma export:
import { prisma } from "@/lib/db";

export const dynamic = "force-dynamic";

export default async function BlogPostPage({
  params,
}: {
  params: { slug: string };
}) {
  const param = params.slug;

  // Try to load by slug
  let post = await prisma.post.findFirst({
    where: { slug: param },
    select: {
      id: true,
      title: true,
      slug: true,
      excerpt: true,
      coverUrl: true,
      contentMd: true,
      createdAt: true,
      author: { select: { name: true } },
    },
  });

  // Fallback if you accidentally linked by id somewhere
  if (!post) {
    post = await prisma.post.findFirst({
      where: { id: param },
      select: {
        id: true,
        title: true,
        slug: true,
        excerpt: true,
        coverUrl: true,
        contentMd: true,
        createdAt: true,
        author: { select: { name: true } },
      },
    });
  }

  if (!post) return notFound();

  return (
    <div className="mx-auto w-full max-w-3xl px-4 py-10">
      <div className="text-sm text-neutral-400">
        <Link href="/blog" className="hover:underline">
          Blog
        </Link>
        <span className="px-2">/</span>
        <span>{post.title}</span>
      </div>

      <header className="mt-6">
        <h1 className="text-4xl font-bold leading-tight">{post.title}</h1>

        <div className="mt-3 text-sm text-neutral-400">
          {new Date(post.createdAt).toLocaleDateString()}
          {post.author?.name ? ` â€¢ ${post.author.name}` : ""}
        </div>

        {post.excerpt ? (
          <p className="mt-6 rounded-2xl border border-white/10 bg-black/20 p-5 text-base text-neutral-100">
            {post.excerpt}
          </p>
        ) : null}
      </header>

      {post.coverUrl ? (
        // eslint-disable-next-line @next/next/no-img-element
        <img
          src={post.coverUrl}
          alt={post.title}
          className="mt-8 h-80 w-full rounded-2xl object-cover"
        />
      ) : null}

      <article className="mt-10 rounded-2xl border border-white/10 bg-black/10 p-6">
        <div className="whitespace-pre-wrap leading-7 text-neutral-100">
          {post.contentMd}
        </div>
      </article>
    </div>
  );
}
